package Clases;

/**
 * La clase Funcionesjuego contiene métodos estáticos que ofrecen funcionalidades específicas para el juego.
 */
public class Funcionesjuego {

    /**
     * Método estático contadorDeTiempo que simula un contador de tiempo para el juego.
     * Este método incrementa un contador de tiempo cada segundo y verifica el estado del jugador.
     * Si el jugador permanece vivo por más de 10 segundos, el estado del jugador se cambia a muerto.
     *
     * @param jugador El objeto Jugador cuyo estado se va a verificar y actualizar.
     */
    public static void contadorDeTiempo(Jugador jugador) {
        int tiempo = 0; // Inicializa el contador de tiempo

        // Bucle que se ejecuta mientras el jugador esté vivo
        while (jugador.isJugadorVivo(jugador) == true) {
            
            tiempo++; // Incrementa el contador de tiempo
            System.out.println("Tiempo: " + tiempo + " segundos"); // Imprime el tiempo transcurrido

            // Verifica si el tiempo ha alcanzado o superado los 10 segundos
            if (tiempo >= 10) {
                jugador.setJugadorMuerto(); // Cambia el estado del jugador a muerto
                System.out.println("El jugador ha muerto."); // Mensaje indicando que el jugador ha muerto
            }

            try {
                Thread.sleep(1000); // Pausa el hilo durante un segundo
            } catch (InterruptedException e) {
                e.printStackTrace(); // Imprime la pila de llamadas si ocurre una interrupción
            }           
        }
    }    
}
